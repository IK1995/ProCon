int ans=0;
  
for(int i=0;i<n;i++){
    
dp[i]=1;
    
for(int j=0;j<i;j++){
      
if(a[j]<a[i]){
        
dp[i]=max(dp[i],dp[j]+1);
      
}
    
}
    
ans=max(ans,dp[i]);
  
}
